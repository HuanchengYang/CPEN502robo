package robot;

import LUT.LUT;
import robocode.*;


import java.awt.*;
import java.awt.geom.Point2D;
import java.io.*;


public class FirstRobot extends AdvancedRobot {
    int dist = 50; // distance to move when we're hit
    public static final int robotMoveDistance = 300;
    public static final double robotTurnDegree =  20.0;
    public static int stateactionIndex[] = new int[9];
    public static final int floor[] = {0,0,0,0,0,0,0,0,0};
    public static final int ceiling[] = {3,19,3,9,9,1,1,1,1};
    public Target target;
    public static LUT lut;
    private static int totalGame;
    private static int winGame;
    private static int loseGame;
    private double firePower;
    private int isHitWall = 0;
    private int isHitByBullet = 0;
    private Learner learner;

    double[] oldstateandaction;
    double[] newstate;
    private double immediatereward;



    /**
     * run:  Fire's main run function
     */
    public void run() {
        totalGame=0;
        winGame=0;
        // Set colors
        setBodyColor(Color.orange);
        setGunColor(Color.orange);
        setRadarColor(Color.red);
        setScanColor(Color.red);
        setBulletColor(Color.red);

        // Spin the gun around slowly... forever
        while (true) {
            lut = new LUT(9,floor,ceiling);
            loadData();
            learner= new Learner(lut);
            target = new Target();
            target.distance = 100000;

            setAdjustGunForRobotTurn(true);
            setAdjustRadarForGunTurn(true);
            turnRadarRightRadians(2 * Math.PI);
            newstate = getState();
            while (true)
            {
                oldstateandaction=robotMovement(newstate,lut);
                firePower = 400/target.distance;
                if (firePower > 3)
                    firePower = 3;
                radarMovement(target);
                gunMovement();
                if (getGunHeat() == 0) {
                    setFire(firePower);
                }
                execute();
                newstate = getState();
                learner.learn(oldstateandaction,newstate,immediatereward);
            }
        }
    }


    private double[] robotMovement(double[] state,LUT lut)
    {

        double[] stateandaction = lut.getMaximumQ(state);
        if (stateandaction[5]==1){
            setAhead(robotMoveDistance);
        }
        if (stateandaction[6]==1) {
            setBack(robotMoveDistance);
        }

        if (stateandaction[7]==1) {
            setAhead(robotMoveDistance);
            setTurnLeft(robotTurnDegree);
        }

        if (stateandaction[8]==1) {
            setAhead(robotMoveDistance);
            setTurnRight(robotTurnDegree);
        }
        return stateandaction;
    }

    private double[] getState()
    {
        double heading = State.getHeading(getHeading());
        double targetDistance = State.getTargetDistance(target.distance);
        double targetBearing = State.getTargetBearing(target.bearing);
        //out.println("Stste(" + heading + ", " + targetDistance + ", " + targetBearing + ", " + isHitWall + ", " + isHitByBullet + ")");
        double[] state = {heading,targetDistance,targetBearing,this.isHitByBullet,this.isHitWall};
        return state;
    }

    private void radarMovement(Target target)
    {
        double radarOffset;
        if (getTime() - target.currentTime > 4) {
            radarOffset = 4*Math.PI;
        } else {

            radarOffset = getRadarHeadingRadians() - (Math.PI/2 - Math.atan2(target.y - getY(),target.x - getX()));

            radarOffset = NormaliseBearing(radarOffset);
            if (radarOffset < 0)
                radarOffset -= Math.PI/10;
            else
                radarOffset += Math.PI/10;
        }
        setTurnRadarLeftRadians(radarOffset);
    }

    private void gunMovement()
    {
        long time;
        long nextTime;
        Point2D.Double p;
        p = new Point2D.Double(target.x, target.y);
        for (int i = 0; i < 20; i++)
        {
            nextTime = (int)Math.round((getrange(getX(),getY(),p.x,p.y)/(20-(3*firePower))));
            time = getTime() + nextTime - 10;
            p = target.guessPosition(time);
        }
        double gunOffset = getGunHeadingRadians() - (Math.PI/2 - Math.atan2(p.y - getY(),p.x -  getX()));
        setTurnGunLeftRadians(NormaliseBearing(gunOffset));
    }

    double NormaliseBearing(double ang) {
        if (ang > Math.PI)
            ang -= 2*Math.PI;
        if (ang < -Math.PI)
            ang += 2*Math.PI;
        return ang;
    }


    public double getrange( double x1,double y1, double x2,double y2 )
    {
        double xo = x2-x1;
        double yo = y2-y1;
        double h = Math.sqrt( xo*xo + yo*yo );
        return h;
    }


    public void onBulletHit(BulletHitEvent e)
    {
        if (target.name == e.getName())
        {

            double change = e.getBullet().getPower() * 9;
            immediatereward += change;
        }
    }


    public void onBulletMissed(BulletMissedEvent e)
    {
        double change = -e.getBullet().getPower();
        immediatereward += change;
    }

    public void onHitByBullet(HitByBulletEvent e)
    {
        if (target.name == e.getName())
        {
            double power = e.getBullet().getPower();
            double change = -(4 * power + 2 * (power - 1));
            immediatereward += change;
        }
        isHitByBullet = 1;
    }

    public void onHitRobot(HitRobotEvent e)
    {
        if (target.name == e.getName())
        {
            double change = -6.0;
            immediatereward += change;
        }
    }

    public void onHitWall(HitWallEvent e)
    {

        double change = -(Math.abs(getVelocity()) * 0.5 - 1);
        immediatereward += change;
        isHitWall = 1;
    }

    /**
     * onScannedRobot: What to do when you see another robot
     */
    public void onScannedRobot(ScannedRobotEvent e)
    {
        if ((e.getDistance() < target.distance)||(target.name == e.getName()))
        {
            double absbearing_rad = (getHeadingRadians()+e.getBearingRadians())%(2*Math.PI);
            target.name = e.getName();
            double h = NormaliseBearing(e.getHeadingRadians() - target.heading);
            h = h/(getTime() - target.currentTime);
            target.changehead = h;
            target.x = getX()+Math.sin(absbearing_rad)*e.getDistance(); 
            target.y = getY()+Math.cos(absbearing_rad)*e.getDistance(); 
            target.bearing = e.getBearingRadians();
            target.heading = e.getHeadingRadians();
            target.currentTime = getTime();			
            target.speed = e.getVelocity();
            target.distance = e.getDistance();
            target.energy = e.getEnergy();
        }
    }

    public void onRobotDeath(RobotDeathEvent e)
    {

        if (e.getName() == target.name)
            target.distance = 10000;
    }

    public void onWin(WinEvent event)
    {
        saveData();
        System.out.println("Win!");
    }

    public void onDeath(DeathEvent event)
    {
        saveData();
        System.out.println("Lose!");
    }

    public void loadData()
    {
        try
        {
            File file = getDataFile("lut3.txt");
            lut.load(file);


        }
        catch (Exception e)
        {
        }
    }

    public void saveData()
    {
        try
        {
            File file = getDataFile("lut3.txt");
            lut.saveData(file);

        }
        catch (Exception e)
        {
            out.println("Exception trying to write: " + e);
        }
    }


}
