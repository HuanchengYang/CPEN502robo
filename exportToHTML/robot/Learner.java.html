package robot;

import LUT.LUT;

public class Learner {
    public static final double LearningRate = 0.1;
    public static final double DiscountRate = 0.9;
    public static final double ExploitationRate = 1;
    private boolean initialize = true;
    private LUT lut;

    public Learner(LUT lut)
    {
        this.lut = lut;
    }

    public void learn(double stateandaction[],double newstate[], double immediateReward)
    {
        double[] oldstate = new double[5];
        immediateReward=0;
        oldstate[0]=stateandaction[0];
        oldstate[1]=stateandaction[1];
        oldstate[2]=stateandaction[2];
        oldstate[3]=stateandaction[3];
        oldstate[4]=stateandaction[4];

        //System.out.println("Reinforcement: " + immediateReward);
        if (initialize)
            initialize = false;
        else
        {
            double oldQValue = lut.outputFor(stateandaction);
            double newQValue = (1 - LearningRate) * oldQValue + LearningRate * (immediateReward + DiscountRate * lut.outputFor(lut.getMaximumQ(oldstate)));
            double convergence = newQValue-oldQValue;
            //System.out.println("convergence is"+ convergence);
            lut.train(stateandaction, newQValue);
        }
    }

}
